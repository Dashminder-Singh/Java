    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
         
 try
 {       
  
 Connection myconnection;
 String path="jdbc:mysql://localhost/";
 String place="student";
 String uname="root";
 String pass="";
 myconnection=DriverManager.getConnection(path+place, uname, pass);
 
 try
 {
 String q="insert into studenttable values(?,?,?,?,?,?,?,?)";
 PreparedStatement mystatement=myconnection.prepareStatement(q);
 
 mystatement.setString(1, studentbox.getText());
 mystatement.setString(2, rollnobox.getText());
 mystatement.setString(3, coursebox.getText());
 mystatement.setString(4, sembox.getText());
 mystatement.setString(5, addressbox.getText());
 
 
 if(malebox.isSelected())
 {
     mystatement.setString(6, "Male");
 }
 else if(femalebox.isSelected())
 {
     mystatement.setString(6, "Female");
 }
 
 SimpleDateFormat myformat=new SimpleDateFormat("yyyy-MM-dd");
 mystatement.setString(7,myformat.format(dobbox.getDate()));
 mystatement.setString(8, phonebox.getText());
 
 
 if(mystatement.executeUpdate()>0)
    {
    JOptionPane.showMessageDialog(rootPane, " Added Successfully");
    studentbox.setText(null);
    rollnobox.setText(null);
    coursebox.setText(null);
    sembox.setText(null);
    addressbox.setText(null);
    buttonGroup1.clearSelection();
    dobbox.setDate(null);
    phonebox.setText(null);
    }
    else
    {
        JOptionPane.showMessageDialog(rootPane, " Not Added Successfully");
    }
  }
    catch(HeadlessException | SQLException e)
    {
    JOptionPane.showMessageDialog(rootPane, "Error in Query " + e.getMessage());
    }
     finally
    {
    myconnection.close();
    }
 }
        catch(HeadlessException | SQLException e)
        {
        JOptionPane.showMessageDialog(rootPane, "Error in Connection " + e.getMessage());
        }
     
    }
